# LUỒNG XỬ LÝ AI - KHÔNG LÀM ĐƠ GIAO DIỆN
# ==========================================
class AIVoiceWorker(QThread):
    progress_signal = pyqtSignal(str)
    finished_signal = pyqtSignal(str)
    play_audio_signal = pyqtSignal(str)
    
    def run(self):
        try:
            start_time = time.time()
            
            # --- 1. RECORD LOCAL AUDIO ---
            fs = 16000  # Sample rate cho Whisper
            duration = 4.0  # Cài đặt ghi âm 4 giây để test
            
            self.progress_signal.emit("🔴 Đang ghi âm (4 giây)... Hãy nói gì đó!")
            myrecording = sd.rec(int(duration * fs), samplerate=fs, channels=1, dtype='int16')
            sd.wait()  # Đợi ghi âm xong
            
            self.progress_signal.emit("⏳ Đang nhận dạng giọng nói bằng Whisper...")
            
            # Lưu tạm vào RAM/File để đưa vào hàm của backend Whisper
            temp_wav = tempfile.mktemp(suffix=".wav")
            with wave.open(temp_wav, 'wb') as wf:
                wf.setnchannels(1)
                wf.setsampwidth(2) # 16-bit
                wf.setframerate(fs)
                wf.writeframes(myrecording.tobytes())
                
            # --- 2. THỰC THI STT (Whisper) ---
            stt_start = time.time()
            segments, info = _stt_model.transcribe(temp_wav, beam_size=5, language="vi")
            text = "".join([segment.text for segment in segments]).strip()
            
            if os.path.exists(temp_wav):
                os.remove(temp_wav)
                
            stt_time = (time.time() - stt_start) * 1000
            
            if not text:
                self.progress_signal.emit("❌ Không nghe rõ âm thanh, màn hình bị nhiễu âm!")
                self.finished_signal.emit("")
                return
                
            self.progress_signal.emit(f"🧑 Bạn: {text} \n(⚡ Tốc độ phân tích âm thanh: {stt_time:.0f}ms)")
            
            # --- 3. THỰC THI LLM (Qwen2 qua llama_index) ---
            self.progress_signal.emit("🤖 LLM đang suy nghĩ câu trả lời...")
            llm_start = time.time()
            # Gọi hàm ask_socrates trực tiếp từ file llm_service.py của bạn
            response = ask_socrates(text)
            llm_time = (time.time() - llm_start) * 1000
            
            # --- 4. THỰC THI TTS (Edge-TTS) ---
            self.progress_signal.emit("🔊 Đang tạo giọng nói (TTS) và phát ra loa...")
            tts_start = time.time()
            # Cố định tên file để rác không bị đẻ ra nhiều
            tts_file = os.path.abspath("temp_reply.mp3") 
            
            import asyncio
            import edge_tts
            
            async def generate_speech():
                communicate = edge_tts.Communicate(response, "vi-VN-HoaiMyNeural")
                await communicate.save(tts_file)
                
            asyncio.run(generate_speech())
            tts_time = (time.time() - tts_start) * 1000
            total_time = (time.time() - start_time) * 1000
            
            self.progress_signal.emit(f"🤖 S-Socrates: {response}")
            self.progress_signal.emit(f"✅ Hoàn thành quy trình: {total_time:.0f}ms (Gồm cả load TTS tốn {tts_time:.0f}ms)\n")
            
            # --- 5. BÁO HIỆU UI PHÁT ÂM THANH Y ĐÚC ỨNG DỤNG ---
            self.play_audio_signal.emit(tts_file)
            
            self.finished_signal.emit("")
            
        except Exception as e:
            self.progress_signal.emit(f"⚠️ Lỗi hệ thống: {str(e)}")
            self.finished_signal.emit("")

# GIAO DIỆN CHÍNH (UI THREAD)
# ==========================================
class TestVoiceUI(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Test Hiệu năng - All In Python App (Real Models)")
        
        self.layout = QVBoxLayout()
        
        self.log_area = QTextEdit()
        self.log_area.setReadOnly(True)
        self.log_area.setStyleSheet("font-size: 16px; background: #000000; color: #ffffff; border-radius: 8px;")
        self.layout.addWidget(self.log_area)
        
        self.btn_voice = QPushButton("🎤 Bấm để thu âm gửi AI")
        self.btn_voice.setStyleSheet("font-size: 20px; font-weight: bold; padding: 25px; background-color: #f44336; color: white; border-radius: 10px;")
        
        self.btn_voice.clicked.connect(self.start_voice_test)
        self.layout.addWidget(self.btn_voice)
        
        widget = QWidget()
        widget.setLayout(self.layout)
        self.setCentralWidget(widget)
        self.resize(600, 500)
        
        # Thiết lập trình phát âm thanh (Media Player) cho TTS
        self.player = QMediaPlayer()
        self.audio_output = QAudioOutput()
        self.player.setAudioOutput(self.audio_output)
        self.audio_output.setVolume(1.0)
    
    def start_voice_test(self):
        self.btn_voice.setText("⏳ Đang nghe & Xử lý Model AI...")
        self.btn_voice.setStyleSheet("font-size: 20px; padding: 25px; background-color: #9e9e9e; color: white; border-radius: 10px;")
        self.btn_voice.setEnabled(False) 
        
        self.log_area.append("\n" + "=" * 50)
        
        self.worker = AIVoiceWorker()
        self.worker.progress_signal.connect(self.on_ai_progress)
        self.worker.finished_signal.connect(self.on_ai_response)
        self.worker.play_audio_signal.connect(self.play_audio)
        self.worker.start()
        
    def on_ai_progress(self, text):
        self.log_area.append(text)
        
    def play_audio(self, file_path):
        self.player.setSource(QUrl.fromLocalFile(file_path))
        self.player.play()
        
    def on_ai_response(self, text):
        if text:
            self.log_area.append(text)
        
        self.btn_voice.setText("🎤 Bấm để thu âm gửi AI")
        self.btn_voice.setStyleSheet("font-size: 20px; font-weight: bold; padding: 25px; background-color: #f44336; color: white; border-radius: 10px;")
        self.btn_voice.setEnabled(True)

if __name__ == "__main__":
    app = QApplication(sys.argv)
    window = TestVoiceUI()
    window.show()
    sys.exit(app.exec())
